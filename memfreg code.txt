<script>
const TOTAL = 64;
const HEADER = 1;
let strategy = 'first-fit';
let memory = [];
let allocations = [];
let nextId = 1;
let stepCount = 0;
let history = { ext:[], int:[], util:[], labels:[] };
let chartFrag, chartUtil;
let totalAllocated = 0;
let totalFreed = 0;

const STRAT_INFO = {
  'first-fit': `Scans the free list from the beginning and uses the <strong style="color:var(--text)">first hole large enough</strong> to satisfy the request. Fast O(n) but tends to leave large holes fragmented near the start of the address space.`,
  'best-fit':  `Searches all free blocks and picks the <strong style="color:var(--text)">smallest hole that fits</strong>. Minimizes wasted space per allocation but produces many tiny unusable gaps over time, increasing external fragmentation.`,
  'worst-fit': `Allocates into the <strong style="color:var(--text)">largest available hole</strong>. Leaves bigger leftover chunks, which can be reused for future allocations, but fragments the address space unevenly.`
};

function initMemory() {
  memory = Array.from({length:TOTAL}, (_,i)=>({idx:i, type:'free', aid:null}));
  allocations=[];nextId=1;stepCount=0;totalAllocated=0;totalFreed=0;
  history={ext:[],int:[],util:[],labels:[]};
}

function getFreeBlocks() {
  const blocks=[];let i=0;
  while(i<TOTAL){
    if(memory[i].type==='free'){
      const s=i;
      while(i<TOTAL&&memory[i].type==='free')i++;
      blocks.push({start:s,size:i-s});
    } else i++;
  }
  return blocks;
}

function pickBlock(needed) {
  const valid = getFreeBlocks().filter(b=>b.size>=needed);
  if(!valid.length) return null;
  if(strategy==='first-fit') return valid[0];
  if(strategy==='best-fit')  return valid.reduce((a,b)=>b.size<a.size?b:a);
  if(strategy==='worst-fit') return valid.reduce((a,b)=>b.size>a.size?b:a);
}

function allocate(reqSize) {
  const needed = reqSize + HEADER;
  const block = pickBlock(needed);
  if(!block){ log(`FAIL: no block â‰¥${needed} units`,'e'); return false; }
  const id = nextId++;
  for(let i=block.start; i<block.start+HEADER; i++) memory[i]={idx:i,type:'header',aid:id};
  for(let i=block.start+HEADER; i<block.start+HEADER+reqSize; i++) memory[i]={idx:i,type:'alloc',aid:id};
  const intFrag = Math.max(0, block.size - needed);
  allocations.push({id,start:block.start,reqSize,intFrag,active:true});
  totalAllocated++;
  log(`ALLOC #${id}  size=${reqSize}  addr=${block.start}  int.frag=${intFrag}`,'a');
  snapshot();
  render();
  return true;
}

function freeById(id) {
  const a = allocations.find(x=>x.id===id&&x.active);
  if(!a) return;
  for(let i=0;i<TOTAL;i++) if(memory[i].aid===id) memory[i]={idx:i,type:'free',aid:null};
  a.active=false; totalFreed++;
  log(`FREE  #${id}  addr=${a.start}  size=${a.reqSize}`,'f');
  snapshot();
  render();
}

function doAlloc() {
  const sz = parseInt(document.getElementById('sz-slider').value);
  allocate(sz);
}

function doFree() {
  const live = allocations.filter(a=>a.active);
  if(!live.length){log('Nothing to free','e');return;}
  const pick = live[Math.floor(Math.random()*live.length)];
  freeById(pick.id);
}

function doBurst() {
  const n = parseInt(document.getElementById('burst-slider').value);
  log(`--- Burst: ${n} ops ---`,'n');
  let i=0;
  const iv = setInterval(()=>{
    if(Math.random()<0.62){ allocate(Math.ceil(Math.random()*7+1)); }
    else { doFree(); }
    if(++i>=n) clearInterval(iv);
  },60);
}

function doCompact() {
  const live = allocations.filter(a=>a.active);
  if(!live.length){log('Nothing to compact','e');return;}
  initMemory();
  const old = [...live];
  allocations=[];nextId=1;
  old.forEach(a=>allocate(a.reqSize));
  log(`Compacted â†’ ${old.length} blocks consolidated`,'n');
}

function doReset() {
  initMemory();
  updateCharts();
  render();
  clearLog();
  log('Simulator initialized','n');
}

function calcStats() {
  const free = getFreeBlocks();
  const live = allocations.filter(a=>a.active);
  const totalFree = free.reduce((s,b)=>s+b.size,0);
  const smallFree = free.filter(b=>b.size<3).reduce((s,b)=>s+b.size,0);
  const intFrag = live.reduce((s,a)=>s+a.intFrag,0);
  const used = live.reduce((s,a)=>s+a.reqSize,0);
  return {
    extPct: Math.round((smallFree/TOTAL)*100),
    intPct: Math.round((intFrag/TOTAL)*100),
    util: Math.round((used/TOTAL)*100),
    totalFree, freeBlocks: free.length, live: live.length
  };
}

function snapshot() {
  stepCount++;
  const s = calcStats();
  history.ext.push(s.extPct);
  history.int.push(s.intPct);
  history.util.push(s.util);
  history.labels.push(stepCount);
  if(history.labels.length>50){
    history.ext.shift();history.int.shift();history.util.shift();history.labels.shift();
  }
  updateCharts();
}

function render() {
  const grid = document.getElementById('mem-grid');
  grid.innerHTML='';
  const tip = document.getElementById('tooltip');
  for(let i=0;i<TOTAL;i++){
    const c = memory[i];
    const div = document.createElement('div');
    div.className='mcell '+(c.type==='free'?'free':c.type==='header'?'header':c.type==='alloc'?'alloc':'free');
    div.addEventListener('mouseenter',e=>{
      let msg = c.type==='free'?`Cell ${i}: free`
              : c.type==='header'?`Cell ${i}: header (alloc #${c.aid})`
              : `Cell ${i}: alloc #${c.aid}`;
      tip.textContent=msg;
      tip.style.opacity='1';
    });
    div.addEventListener('mousemove',e=>{ tip.style.left=(e.clientX+12)+'px'; tip.style.top=(e.clientY-30)+'px'; });
    div.addEventListener('mouseleave',()=>{ tip.style.opacity='0'; });
    if(c.type==='alloc'&&c.aid) div.addEventListener('click',()=>freeById(c.aid));
    grid.appendChild(div);
  }

  // internal frag highlight
  const live = allocations.filter(a=>a.active);
  live.forEach(a=>{
    const endData = a.start+HEADER+a.reqSize;
    const endBlock = a.start + HEADER + a.reqSize + a.intFrag;
    for(let i=endData;i<Math.min(endBlock,TOTAL)&&i<endBlock;i++){
      if(memory[i].type==='free'){
        const c2 = grid.children[i];
        if(c2){ c2.className='mcell intfrag'; }
      }
    }
  });

  const s = calcStats();
  const setM = (id,val,cls='neutral')=>{
    const el = document.getElementById(id);
    el.textContent=val;
    el.className='metric-cell-val '+cls;
  };
  const extCls = s.extPct>30?'bad':s.extPct>12?'warn':'good';
  const intCls = s.intPct>25?'bad':s.intPct>12?'warn':'good';
  const uCls   = s.util>65?'good':s.util>35?'warn':'bad';
  const fbCls  = s.freeBlocks>10?'bad':s.freeBlocks>5?'warn':'good';
  setM('m-ext', s.extPct+'%', extCls);
  setM('m-int', s.intPct+'%', intCls);
  setM('m-util', s.util+'%', uCls);
  setM('m-free', s.totalFree+'u', 'info');
  setM('m-fblocks', s.freeBlocks, fbCls);
  setM('m-live', s.live, 'neutral');
}
